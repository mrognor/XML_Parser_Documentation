152 template<class T>
153 bool Validate(T begin, T end, std::list<XmlData>* listWithAllData = nullptr, bool isPathSaving = true)
154 {
155 if (listWithAllData != nullptr)
156 listWithAllData->clear();
157
158 // String to store all chars in tags
159 std::string tagString;
160 // String to store all chars between tags
161 std::string betweenTagsString;
162 // String to store comments
163 std::string commentString;
164 // List with all opened but not closed tags
165 std::list<std::string> tagStringsStackList;
166 // Is string with tag name and params opened by "<" symbol
167 bool isTagStringOpened = false;
168 // Line number in file or number of element in vector
169 int lineNumber = 0;
170 // Is string with comment
171 bool isComment = false;
172 // Variable to store if it was root element
173 bool wasAtLeastOneTag = false;
174 // Is string with xml specification
175 bool isSpecString = false;
176
177 // For loop by iterator
178 for (auto xmlStringIt = begin; xmlStringIt != end; xmlStringIt++)
179 {
180 lineNumber++;
181 std::string xmlString = (std::string)(*xmlStringIt);
182
183 // Ignore line with xml specification
184 if (lineNumber == 1)
185 {
186 std::string trimString = TrimString(xmlString, false);
187 // Check if it is starts from "<?" string
188 if(xmlString.length() >= 5 && xmlString[0] == '<' && xmlString[1] == '?' && xmlString[2] == 'x' &&
189 xmlString[3] == 'm' && xmlString[4] == 'l')
190 {
191 isSpecString = true;
192 continue;
193 }
194 }
195
196 // For loop by string
197 for (auto stringChar = xmlString.begin(); stringChar != xmlString.end(); stringChar++)
198 {
199 // Check if it is closing spec string
200 if (isSpecString && *stringChar == '>' && *(stringChar - 1) == '?')
201 {
202 isSpecString = false;
203 continue;
204 }
205
206 // Skip logic if it is spec string
207 if (isSpecString)
208 continue;
209
210 // Check if it is opening comment
211 if ((stringChar + 1) != xmlString.end() && (stringChar + 2) != xmlString.end() &&
212 *stringChar == '<' && *(stringChar + 1) == '!' && *(stringChar + 2) == '-' && *(stringChar + 3) == '-')
213 {
214 isComment = true;
215 commentString += *stringChar;
216 continue;
217 }
218
219 // Check if it is closing comment
220 if (stringChar != xmlString.begin() && (stringChar - 1) != xmlString.begin() &&
221 *stringChar == '>' && *(stringChar - 1) == '-' && *(stringChar - 2) == '-')
222 {
223 isComment = false;
224 commentString += *stringChar;
225
226 // Check if it is string "--" on comment
227 if (commentString.substr(4, commentString.length() - 7).find("--") != -1)
228 {
229 LOG("The string \"--\" is not valid inside comments");
230 if (listWithAllData != nullptr)
231 listWithAllData->clear();
232 return false;
233 }
234
235 // Add comment to XmlData list
236 if(listWithAllData != nullptr && !commentString.empty())
237 {
238 // Create XmlData object
239 XmlData commentData;
240 commentData.Data = commentString;
241 commentData.DataType = comment;
242
243 // Find path
244 if (isPathSaving)
245 {
246 std::string path = "/";
247 for (const auto& it : tagStringsStackList)
248 path = path + it + "/";
249 commentData.Path = path;
250 }
251
252 // Store data
253 listWithAllData->push_back(commentData);
254 }
255
256 commentString.clear();
257 continue;
258 }
259
260 // Skip all logic if it is a comment
261 if (isComment)
262 {
263 commentString += *stringChar;
264 continue;
265 }
266
267 // Check opening tag string
268 if (*stringChar == '<')
269 {
270 // Check if it is second root element
271 if (tagStringsStackList.size() == 0 && wasAtLeastOneTag)
272 {
273 LOG("There can be only one root element in a file. New root element on line: " + std::to_string(lineNumber));
274 if (listWithAllData != nullptr)
275 listWithAllData->clear();
276 return false;
277 }
278
279 // '<' symbol without closing tag
280 if (isTagStringOpened)
281 {
282 LOG("Opened new tag without closing previous on line: " + std::to_string(lineNumber));
283 if (listWithAllData != nullptr)
284 listWithAllData->clear();
285 return false;
286 }
287
288 isTagStringOpened = true;
289
290 // Check string between tags to valid
291 std::string trimmedValueString = TrimString(betweenTagsString);
292 if (!CheckString(trimmedValueString))
293 {
294 LOG("Invalid symbol on line: " + std::to_string(lineNumber));
295 if (listWithAllData != nullptr)
296 listWithAllData->clear();
297 return false;
298 }
299
300 // Add string between tag to XmlData list
301 if (!TrimString(betweenTagsString).empty() && listWithAllData != nullptr)
302 {
303 // Create XmlData object
304 XmlData textData;
305 textData.Data = TrimString(betweenTagsString);
306 textData.DataType = text;
307
308 // Find path
309 if (isPathSaving)
310 {
311 std::string path = "/";
312 for (const auto& it : tagStringsStackList)
313 path = path + it + "/";
314 textData.Path = path;
315 }
316
317 // Store data
318 listWithAllData->push_back(textData);
319 }
320
321 betweenTagsString.clear();
322 continue;
323 }
324
325 // Check closing tag string
326 if (*stringChar == '>')
327 {
328 isTagStringOpened = false;
329 std::string tagName;
330 std::map<std::string, std::string> paramsAndValues;
331
332 // Check if it is closing tag
333 if (tagString[0] == '/')
334 {
335 // Check if closing tag equals or not opening tag
336 if (tagStringsStackList.size() != 0 && TrimString(tagString.substr(1)) == tagStringsStackList.back())
337 {
338 tagName = tagStringsStackList.back();
339 tagStringsStackList.pop_back();
340 }
341 else
342 {
343 if (tagStringsStackList.empty())
344 {
345 LOG("Closing not opened tag on line: " + std::to_string(lineNumber));
346 }
347 else
348 {
349 LOG("Closing wrong tag on line: " + std::to_string(lineNumber) + " Expected: " + tagStringsStackList.back());
350 }
351 if (listWithAllData != nullptr)
352 listWithAllData->clear();
353 return false;
354 }
355 }
356 else
357 {
358 // Check if it is not one line tag
359 if (tagString[tagString.length() - 1] != '/')
360 {
361 if (!ParseTagString(tagString, tagName, paramsAndValues))
362 {
363 LOG("Wrong tag name on line: " + std::to_string(lineNumber));
364 if (listWithAllData != nullptr)
365 listWithAllData->clear();
366 return false;
367 }
368
369 tagStringsStackList.push_back(tagName); // Add opening tag to stack
370 wasAtLeastOneTag = true;
371 }
372 else // One line tag check
373 {
374 if (!ParseTagString(tagString.substr(0, tagString.length() - 1), tagName, paramsAndValues))
375 {
376 LOG("Wrong tag name on line: " + std::to_string(lineNumber));
377 if (listWithAllData != nullptr)
378 listWithAllData->clear();
379 return false;
380 }
381 }
382 }
383
384 // Add tag to XmlDataList
385 if (listWithAllData != nullptr)
386 {
387 // Create XmlData object
388 XmlData tagData;
389
390 // Find XmlData object type
391 // Opening tag
392 tagData.DataType = openingTag;
393 // Closing tag
394 if (tagString[0] == '/')
395 tagData.DataType = closingTag;
396 // Inline tag
397 if (tagString[tagString.length() - 1] == '/')
398 tagData.DataType = inlineTag;
399
400 // Format raw data
401 if (tagData.DataType == closingTag)
402 tagData.Data = "<" + TrimString(tagString) + ">";
403 else
404 {
405 tagData.Data = "<" + tagName;
406
407 for (const auto& it : paramsAndValues)
408 tagData.Data = tagData.Data + " " + it.first + "=\"" + it.second + "\"";
409
410 if (tagData.DataType == inlineTag)
411 tagData.Data += "/";
412
413 tagData.Data += ">";
414 }
415
416 // Set XmlData object tag name and map with params and values
417 tagData.ParamsAndValues = paramsAndValues;
418 tagData.TagName = tagName;
419
420 // Find path
421 if (isPathSaving)
422 {
423 std::string path = "/";
424 if (!tagStringsStackList.empty())
425 {
426 // Add all path elements except last
427 for (auto it = tagStringsStackList.begin(); it != --tagStringsStackList.end(); it++)
428 path = path + *it + "/" ;
429
430 // Add last path element if it is closing tag
431 if (tagData.DataType == closingTag || tagData.DataType == inlineTag)
432 path = path + tagStringsStackList.back() + "/";
433 }
434 tagData.Path = path;
435 }
436
437 // Store data
438 listWithAllData->push_back(tagData);
439 }
440 tagString.clear();
441 continue;
442 }
443
444 // Check if tag line opened
445 if (isTagStringOpened)
446 tagString += *stringChar;
447 else betweenTagsString += *stringChar;
448 }
449
450 // Skip logic if it is spec string
451 if (isSpecString)
452 continue;
453
454 // Add new line symbol to comment
455 if (isComment)
456 {
457 commentString += "\n";
458 continue;
459 }
460
461 // Add non tag and non comment line to data list
462 if (!isTagStringOpened && !isComment && !TrimString(betweenTagsString).empty())
463 {
464 if (tagStringsStackList.size() == 0)
465 {
466 LOG("Text outside xml root on line number: " + std::to_string(lineNumber));
467 if (listWithAllData != nullptr)
468 listWithAllData->clear();
469 return false;
470 }
471
472 // Add string between tag to XmlData list
473 if (!TrimString(betweenTagsString).empty() && listWithAllData != nullptr)
474 {
475 // Create XmlData object
476 XmlData textData;
477 textData.Data = TrimString(betweenTagsString);
478 textData.DataType = text;
479
480 // Find path
481 if (isPathSaving)
482 {
483 std::string path = "/";
484 for (const auto& it : tagStringsStackList)
485 path = path + it + "/";
486 textData.Path = path;
487 }
488
489 // Store data
490 listWithAllData->push_back(textData);
491 }
492
493 betweenTagsString.clear();
494 }
495 }
496
497 // Check if it is some unclosed tags
498 if (!tagStringsStackList.empty())
499 {
500 LOG("Dont closed tags")
501 if (listWithAllData != nullptr)
502 listWithAllData->clear();
503 return false;
504 }
505 else
506 return true;
507 }
